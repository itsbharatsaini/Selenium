pipeline {
     agent any

     stage ('Git  Clone') {
            steps {
                git branch: 'main', credentialsId: 'GitHub_Credentials', url: 'https://github.com/itsbharatsaini/Selenium.git'
            }
        }
       
        stage ('kubernetes Deploy') {
			steps {
				script {
					kubernetesDeploy(
						configs: 'selenium-pod.yaml',
						kubeconfigId: 'K8S'
						)           
				   
				}
			}
		}
  
    }  
}
